32 // Helper that reserves space in a container based on the number of elements in a sequence
33 // if it's possible to do so quickly (e.g. with random-access iterators or size())
34 template<typename C, typename Seq>
35 auto try_reserve(C& cnt, const Seq& seq) -> typename std::enable_if<is_enumerable<Seq>::value, bool>::type
36 {
37  const bool can_reserve = seq.has_fast_size();
38  if (can_reserve) {
39  cnt.reserve(seq.size());
40  }
41  return can_reserve;
42 }
43 template<typename C, typename Seq>
44 auto try_reserve(C& cnt, const Seq& seq) -> typename std::enable_if<!is_enumerable<Seq>::value &&
45  coveo::detail::has_size_const_method<Seq>::value, bool>::type
46 {
47  cnt.reserve(seq.size());
48  return true;
49 }
50 template<typename C, typename Seq>

69 // Helper that returns a size_delegate for a sequence if it's possible to quickly calculate
70 // its size (e.g. with random-access iterators or size())
71 template<typename Seq>
72 auto try_get_size_delegate(const Seq& seq) -> typename std::enable_if<is_enumerable<Seq>::value,
73  std::function<std::size_t()>>::type
74 {
75  std::function<std::size_t()> siz;
76  if (seq.has_fast_size()) {
77  const std::size_t size = seq.size();
78  siz = [size]() -> std::size_t { return size; };
79  }
80  return siz;
81 }

85  std::function<std::size_t()>>::type
