59 /**
60  * @brief Helper function to quickly reserve space in a container if possible.
61  * @headerfile linq_util.h <coveo/linq/linq_util.h>
62  *
63  * Attempts to @c reserve space in container @c cnt to hold as many
64  * elements as found in sequence @c seq. This is performed only if
65  * it's possible to do so "quickly".
66  *
67  * - If @c seq is a <tt>coveo::enumerable</tt>, space is reserved
68  * by using its <tt>coveo::enumerable::size()</tt> method if
69  * <tt>coveo::enumerable::has_fast_size()</tt> returns @c true.
70  * - If @c seq is a container with a @c size method, space is
71  * reserved by using that method.
72  * - If @c seq is a sequence that uses random-access iterators,
73  * space is reserved by using <tt>std::distance()</tt>.
74  * - Otherwise, space is not reserved.
75  *
76  * @param cnt Container in which to reserve space.
77  * @param seq Sequence to use to try to reserve space for.
78  * @return @c true if space has actually been reserved in @c cnt.
79  */
80 template<typename C, typename Seq>
81 auto try_reserve(C& cnt, const Seq& seq) -> typename std::enable_if<is_enumerable<Seq>::value, bool>::type
82 {
83  const bool can_reserve = seq.has_fast_size();
84  if (can_reserve) {
85  cnt.reserve(seq.size());
86  }
87  return can_reserve;
88 }

91  coveo::detail::has_size_const_method<Seq>::value, bool>::type
92 {
93  cnt.reserve(seq.size());
94  return true;
95 }

115 /**
116  * @brief Helper function to get a fast size delegate if possible.
117  * @headerfile linq_util.h <coveo/linq/linq_util.h>
118  *
119  * Attempts to create a <tt>coveo::enumerable::size_delegate</tt>
120  * that can quickly calculate the number of elements found in
121  * sequence @c seq. A size delegate is returned only if it's
122  * possible to calculate the number of elements "quickly".
123  *
124  * - If @c seq is a <tt>coveo::enumerable</tt>, its
125  * <tt>coveo::enumerable::size()</tt> method is used to produce
126  * the size delegate if <tt>coveo::enumerable::has_fast_size()</tt>
127  * returns @c true.
128  * - If @c seq is a container with a @c size method, a size delegate
129  * is produced using that method.
130  * - If @c seq is a sequence that uses random-access iterators,
131  * a size delegate is produced by using <tt>std::distance()</tt>.
132  * - Otherwise, no size delegate is produced.
133  *
134  * @param seq Sequence to calculate the number of elements of
135  * in order to produce the size delegate.
136  * @return <tt>coveo::enumerable::size_delegate</tt> instance,
137  * or @c nullptr if it's not possible to quickly calculate
138  * the number of elements in @c seq.
139  * @see coveo::enumerable::size_delegate
140  */
141 template<typename Seq>
142 auto try_get_size_delegate(const Seq& seq) -> typename std::enable_if<is_enumerable<Seq>::value,
143  std::function<std::size_t()>>::type
144 {
145  std::function<std::size_t()> siz;
146  if (seq.has_fast_size()) {
147  const std::size_t size = seq.size();
148  siz = [size]() -> std::size_t { return size; };
149  }
150  return siz;
151 }
152 template<typename Seq>

155  std::function<std::size_t()>>::type
156 {
157  const std::size_t size = seq.size();
158  return [size]() -> std::size_t { return size; };
159 }
