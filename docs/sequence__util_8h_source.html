27  * @brief Traits class for elements in a sequence.
28  * @headerfile sequence_util.h <coveo/seq/sequence_util.h>
29  *
30  * Traits class containing definitions pertaining to the elements of a sequence.
31  * For sequences of references or <tt>std::reference_wrapper</tt>s, provides
32  * information about the referred type instead.
33  *
34  * @tparam T Type of elements in the sequence.
35  */
36 template<typename T>
37 struct seq_element_traits
38 {
39  /**
40  * @brief Type of element in the sequence.
41  *
42  * Type of the sequence's elements. Corresponds to @c T.
43  */
44  using value_type = T;
45 
46  /**
47  * @brief Type of element in the sequence, but @c const.
48  *
49  * Same as <tt>coveo::seq_element_traits::value_type</tt>, but @c const.
50  */
51  using const_value_type = const value_type;
52 
53  /**
54  * @brief Raw type of element in the sequence.
55  *
56  * Same as <tt>coveo::seq_element_traits::value_type</tt>, but "raw", e.g. without @c const or @c volatile.
57  */
58  using raw_value_type = typename std::remove_cv<value_type>::type;
59 
60  /**
61  * @brief Pointer to a sequence element.
62  *
63  * Pointer to an element in the sequence.
64  * Corresponds to <tt>coveo::seq_element_traits::value_type*</tt>.
65  */
66  using pointer = value_type*;
67 
68  /**
69  * @brief Reference to a sequence element.
70  *
71  * Reference to an element in the sequence.
72  * Corresponds to <tt>coveo::seq_element_traits::value_type&</tt>.
73  */
74  using reference = value_type&;
75 
76  /**
77  * @brief Pointer to a @c const sequence element.
78  *
79  * Pointer to a @c const element in the sequence.
80  * Corresponds to <tt>coveo::seq_element_traits::const_value_type*</tt>.
81  */
82  using const_pointer = const_value_type*;
83 
84  /**
85  * @brief Reference to a @c const sequence element.
86  *
87  * Reference to a @c const element in the sequence.
88  * Corresponds to <tt>coveo::seq_element_traits::const_value_type&</tt>.
89  */
90  using const_reference = const_value_type&;
91 };

98 /**
99  * @brief Traits class for a sequence.
100  * @headerfile sequence_util.h <coveo/seq/sequence_util.h>
101  *
102  * Traits class containing definitions pertaining to a sequence. A shorthand
103  * for <tt>coveo::seq_element_traits</tt> that infers the sequence's @c value_type
104  * from the return value of its iterators. Also provides the type of iterator
105  * used by the sequence.
106  *
107  * @tparam Seq Type of sequence.
108  */
109 template<typename Seq>
110 struct seq_traits : public seq_element_traits<decltype(*std::begin(std::declval<Seq&>()))>
111 {
112  /**
113  * @brief Type of iterator used by the sequence.
114  *
115  * Type of iterator used by the sequence, which is defined as the type
116  * of iterator returned when calling <tt>std::begin()</tt>.
117  */
118  using iterator_type = typename std::decay<decltype(std::begin(std::declval<Seq&>()))>::type;
119 };
