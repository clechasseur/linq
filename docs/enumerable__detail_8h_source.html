18 // Type trait that can be used to know if std::begin(T) is valid.
19 // Detects both std::begin specializations and begin methods.
20 template<typename T>
21 class has_begin
22 {
23  static_assert(sizeof(std::int_least8_t) != sizeof(std::int_least32_t),
24  "has_begin only works if int_least8_t has a different size than int_least32_t");
25 
26  template<typename C> static std::int_least8_t test(decltype(std::begin(std::declval<C>()))*); // Will be selected if std::begin(C) works
27  template<typename C> static std::int_least32_t test(...); // Will be selected otherwise
28 public:
29  static constexpr bool value = sizeof(test<T>(nullptr)) == sizeof(std::int_least8_t);
